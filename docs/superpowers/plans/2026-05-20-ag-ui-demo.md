# AG-UI Demo Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fullstack demo showcasing Microsoft Agent Framework multi-agent HandoffBuilder workflow with AG-UI protocol streaming, using DeepSeek LLM and a React+Vite frontend.

**Architecture:** FastAPI backend exposes an AG-UI SSE endpoint via `add_agent_framework_fastapi_endpoint`, wrapping a `HandoffBuilder` workflow with 4 agents (Triage, Weather, Hotel, Summary). React frontend uses `@ag-ui/client` `HttpAgent` to connect directly to the SSE endpoint and renders streaming messages, tool call status, and an itinerary card.

**Tech Stack:** Python (uv, agent-framework, FastAPI, DeepSeek/OpenAI API), React 18 + TypeScript + Vite + Tailwind CSS + `@ag-ui/client`

```python
# backend/src/workflow.py
from agent_framework import Message
from agent_framework.orchestrations import HandoffBuilder

from .agents import triage_agent, weather_agent, hotel_agent, summary_agent


def _termination_condition(conversation: list[Message]) -> bool:
    """当 Summary Agent 输出结束标记时终止工作流。"""
    for message in reversed(conversation):
        if message.role != "assistant":
            continue
        text = message.text or ""
        if "旅行计划已生成" in text or "祝您旅途愉快" in text:
            return True
    return False


def create_travel_workflow():
    """构建旅行规划 Handoff 工作流。"""
    builder = HandoffBuilder(
        name="travel_planner_workflow",
        participants=[triage_agent, weather_agent, hotel_agent, summary_agent],
        termination_condition=_termination_condition,
    )
    (
        builder
        .add_handoff(
            triage_agent,
            [weather_agent],
            description="用户需要查询天气",
        )
        .add_handoff(
            triage_agent,
            [hotel_agent],
            description="用户需要查询酒店",
        )
        .add_handoff(
            triage_agent,
            [summary_agent],
            description="信息收集完毕，生成旅行计划",
        )
        .add_handoff(
            weather_agent,
            [triage_agent],
            description="天气查询完成，返回调度",
        )
        .add_handoff(
            hotel_agent,
            [triage_agent],
            description="酒店查询完成，返回调度",
        )
        .add_handoff(
            summary_agent,
            [triage_agent],
            description="计划生成完毕",
        )
    )
    return builder.with_start_agent(triage_agent).build()
```

```python
# backend/src/main.py
import logging
import os
from contextlib import asynccontextmanager

import uvicorn
from agent_framework.ag_ui import AgentFrameworkWorkflow, add_agent_framework_fastapi_endpoint
from dotenv import load_dotenv
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from .workflow import create_travel_workflow

load_dotenv()

logger = logging.getLogger(__name__)


def create_app() -> FastAPI:
    app = FastAPI(
        title="旅行规划 AG-UI Demo",
        description="Microsoft Agent Framework + AG-UI 多 Agent 工作流演示",
        version="0.1.0",
    )

    cors_origins = [
        origin.strip()
        for origin in os.getenv("CORS_ORIGINS", "http://localhost:5173").split(",")
        if origin.strip()
    ]
    app.add_middleware(
        CORSMiddleware,
        allow_origins=cors_origins,
        allow_credentials=True,
        allow_methods=["*"],
        allow_headers=["*"],
    )

    travel_workflow = AgentFrameworkWorkflow(
        workflow_factory=lambda _thread_id: create_travel_workflow(),
        name="travel_planner_workflow",
        description="旅行规划多 Agent 工作流演示",
    )

    add_agent_framework_fastapi_endpoint(
        app=app,
        agent=travel_workflow,
        path="/api/agent",
    )

    @app.get("/healthz")
    async def healthz() -> dict[str, str]:
        return {"status": "ok"}

    return app


app = create_app()


def main() -> None:
    logging.basicConfig(level=logging.INFO)
    host = os.getenv("HOST", "127.0.0.1")
    port = int(os.getenv("PORT", "8000"))
    print(f"AG-UI demo backend running at http://{host}:{port}")
    print(f"AG-UI endpoint: POST /api/agent")
    uvicorn.run(app, host=host, port=port)


if __name__ == "__main__":
    main()
```

```typescript
// frontend/src/components/TravelPlanner.tsx
import { useCallback, useEffect, useRef, useState } from "react";
import { HttpAgent } from "@ag-ui/client";
import type {
  RunStartedEvent,
  TextMessageContentEvent,
  TextMessageEndEvent,
  TextMessageStartEvent,
  ToolCallResultEvent,
  ToolCallStartEvent,
  RunFinishedEvent,
} from "@ag-ui/client";
import type { AgentStatus, ChatMessage, Itinerary } from "../types";
import AgentStatusBar from "./AgentStatus";
import ChatPanel from "./ChatPanel";
import ItineraryCard from "./ItineraryCard";

const AGENT_URL = import.meta.env.VITE_AGENT_URL || "/api/agent";

export default function TravelPlanner() {
  const [messages, setMessages] = useState<ChatMessage[]>([]);
  const [status, setStatus] = useState<AgentStatus>("idle");
  const [itinerary, setItinerary] = useState<Itinerary>({});
  const [currentResponse, setCurrentResponse] = useState("");
  const [activeTool, setActiveTool] = useState<string | null>(null);
  const agentRef = useRef<HttpAgent | null>(null);
  const messagesRef = useRef<ChatMessage[]>([]);

  useEffect(() => {
    agentRef.current = new HttpAgent({ url: AGENT_URL });
  }, []);

  // Keep ref in sync with state to avoid stale closures
  useEffect(() => {
    messagesRef.current = messages;
  }, [messages]);

  const handleSend = useCallback(
    async (userText: string) => {
      if (!agentRef.current) return;

      const userMsg: ChatMessage = {
        id: crypto.randomUUID(),
        role: "user",
        content: userText,
      };
      setMessages((prev) => [...prev, userMsg]);
      setStatus("thinking");
      setCurrentResponse("");
      setActiveTool(null);

      try {
        const stream = await agentRef.current.run({
          messages: [
            ...messagesRef.current.map((m) => ({
              role: m.role,
              content: m.content,
            })),
            { role: "user", content: userText },
          ],
        });

        let assistantContent = "";

        for await (const event of stream) {
          switch (event.type) {
            case "RUN_STARTED": {
              setStatus("thinking");
              break;
            }
            case "TEXT_MESSAGE_START": {
              assistantContent = "";
              break;
            }
            case "TEXT_MESSAGE_CONTENT": {
              assistantContent += event.content || "";
              setCurrentResponse(assistantContent);
              break;
            }
            case "TEXT_MESSAGE_END": {
              // Commit each completed assistant message immediately
              if (assistantContent) {
                setMessages((prev) => [
                  ...prev,
                  {
                    id: crypto.randomUUID(),
                    role: "assistant",
                    content: assistantContent,
                  },
                ]);
                setCurrentResponse("");
                assistantContent = "";
              }
              break;
            }
            case "TOOL_CALL_START": {
              setStatus("tool_call");
              setActiveTool((event as ToolCallStartEvent).toolName || null);
              break;
            }
            case "TOOL_CALL_RESULT": {
              const tcEvent = event as ToolCallResultEvent;
              try {
                const result = JSON.parse(tcEvent.result || "{}");
                const toolName = tcEvent.toolName || "";
                if (toolName === "get_weather" && result.city) {
                  setItinerary((prev) => ({
                    ...prev,
                    city: result.city,
                    weather: result,
                  }));
                } else if (toolName === "search_hotels" && Array.isArray(result)) {
                  setItinerary((prev) => ({
                    ...prev,
                    hotels: result,
                  }));
                }
              } catch {
                // ignore non-JSON results
              }
              setStatus("thinking");
              setActiveTool(null);
              break;
            }
            case "RUN_FINISHED": {
              setStatus("idle");
              // Commit any remaining content
              if (assistantContent) {
                setMessages((prev) => [
                  ...prev,
                  {
                    id: crypto.randomUUID(),
                    role: "assistant",
                    content: assistantContent,
                  },
                ]);
                setCurrentResponse("");
              }
              break;
            }
          }
        }
      } catch (err) {
        setStatus("error");
        console.error("Agent error:", err);
      }
    },
    []
  );

  return (
    <div className="max-w-4xl mx-auto p-4 flex flex-col gap-4 h-screen">
      <h1 className="text-2xl font-bold text-gray-800">旅行规划助手</h1>
      <AgentStatusBar status={status} activeTool={activeTool} />
      <div className="flex-1 flex gap-4 overflow-hidden">
        <div className="flex-1 flex flex-col overflow-hidden">
          <ChatPanel
            messages={messages}
            currentResponse={currentResponse}
            onSend={handleSend}
          />
        </div>
        <div className="w-80">
          <ItineraryCard itinerary={itinerary} />
        </div>
      </div>
    </div>
  );
}
```

```typescript
// frontend/src/components/ChatPanel.tsx
import { useState } from "react";
import type { ChatMessage } from "../types";

interface ChatPanelProps {
  messages: ChatMessage[];
  currentResponse: string;
  onSend: (text: string) => void;
}

export default function ChatPanel({
  messages,
  currentResponse,
  onSend,
}: ChatPanelProps) {
  const [input, setInput] = useState("");

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!input.trim()) return;
    onSend(input.trim());
    setInput("");
  };

  return (
    <div className="flex flex-col h-full border rounded-lg bg-white">
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {messages.map((msg) => (
          <div
            key={msg.id}
            className={`flex ${
              msg.role === "user" ? "justify-end" : "justify-start"
            }`}
          >
            <div
              className={`max-w-[70%] rounded-lg px-4 py-2 ${
                msg.role === "user"
                  ? "bg-blue-500 text-white"
                  : "bg-gray-100 text-gray-800"
              }`}
            >
              <p className="whitespace-pre-wrap">{msg.content}</p>
            </div>
          </div>
        ))}
        {currentResponse && (
          <div className="flex justify-start">
            <div className="max-w-[70%] rounded-lg px-4 py-2 bg-gray-100 text-gray-800">
              <p className="whitespace-pre-wrap">{currentResponse}</p>
              <span className="inline-block w-2 h-4 bg-gray-400 animate-pulse ml-1" />
            </div>
          </div>
        )}
      </div>
      <form
        onSubmit={handleSubmit}
        className="border-t p-3 flex gap-2"
      >
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="说点什么..."
          className="flex-1 border rounded-lg px-3 py-2 focus:outline-none focus:ring-2 focus:ring-blue-500"
        />
        <button
          type="submit"
          className="bg-blue-500 text-white px-4 py-2 rounded-lg hover:bg-blue-600"
        >
          发送
        </button>
      </form>
    </div>
  );
}
```

```bash
cd backend
uv sync
```

Verify `.env` has correct `DEEPSEEK_API_KEY`. Test with:
```bash
curl https://api.deepseek.com/v1/models \
  -H "Authorization: Bearer $DEEPSEEK_API_KEY"
```

```bash
cd frontend
npm list @ag-ui/client
```
